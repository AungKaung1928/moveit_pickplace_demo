```bash

git clone <repository-url> simple_moveit_demo

# Install dependencies
cd ~/moveit_project_ws
rosdep install --from-paths src --ignore-src -r -y

# Build
colcon build --packages-select simple_moveit_demo
source install/setup.bash
```
